left_click.normal:
  interactable ->
    interact / move_then_interact

  enemy ->
    hard target
    approach if skill policy/range requires it
    otherwise use skill

  empty ground ->
    move


left_click.shift:
  interactable ->
    ignore hovered interactable
    attack_in_place

  enemy ->
    depends on skill policy

    melee-style:
      ignore hovered enemy
      attack_in_place
        soft acquire if possible
        otherwise attack toward cursor

    ranged-style:
      hard target enemy
      no approach
      use skill on enemy while held
      if enemy dies, stop until release

  empty ground ->
    attack_in_place


skill_button.normal:
  interactable ->
    depends on skill policy

    melee-style:
      interact / move_then_interact

    ranged-style:
      ignore interactable
      attack_in_place / cast toward cursor

  enemy ->
    hard target
    approach if policy/range says to
    otherwise use skill

  empty ground ->
    depends on skill policy

    melee-style:
      move_with_soft_skill_use

    ranged-style:
      attack_in_place / cast toward cursor


skill_button.shift:
